                driving_info = DistanceSearch.run_for_all_cities('Atlanta', cities, [city_0, city_1])  # use driving_info, do not customize
                driving_info = DistanceSearch.run_for_all_cities(city_0, cities, [city_1])  # use driving_info, do not customize
                driving_info = DistanceSearch.run_for_all_cities(city_1, cities, ['Atlanta'])  # use driving_info, do not customize
                
                # Calculate self-driving and taxi prices based on distance for 1 person  
                self_driving_0_price = 0.05 * driving_0_distance * math.ceil(1 / 5)
                taxi_0_price = driving_0_distance * math.ceil(1 / 4)

                driving_1_arrtime, _ = DistanceSearch.get_info(driving_info, city_0, city_1, 'Duration') # Use driving_1_arrtime, do not customize
                driving_2_arrtime, _ = DistanceSearch.get_info(driving_info, city_1, 'Atlanta', 'Duration') # Use driving_2_arrtime, do not customize

                transportation_1_arrtime = If(variables['flight'][1], flight_1_arrtime, driving_1_arrtime)

                restaurant_price, restaurant_cuisines = RestaurantSearch.run_for_all_cities(cities)

                variables['restaurant_in_which_city'] = [Int('restaurant_' + str(i)) for i in range(3*7)]

                for i, variable in enumerate(variables['restaurant_in_which_city']):
                    date_index = i // 3
                    meal_index = i % 3
                    if meal_index == 0: # breakfast
                        s.assert_and_track(Or(variable == city_list[date_index], variable == city_list[date_index+1]),  'eat in which city b')
                        s.assert_and_track(Implies(arrives[date_index]> 10, variable == city_list[date_index]),'eat in which city b')
                        s.assert_and_track(Implies(arrives[date_index]< 5, variable == city_list[date_index+1]),'eat in which city b')

                        s.assert_and_track(Or(variable == city_list[date_index], variable == city_list[date_index+1]),  'eat in which city l')
                        s.assert_and_track(Implies(arrives[date_index]> 15, variable == city_list[date_index]),'eat in which city l')
                        s.assert_and_track(Implies(arrives[date_index]< 10, variable == city_list[date_index+1]),'eat in which city l')
                    if meal_index == 2: # dinner
                        s.assert_and_track(Or(variable == city_list[date_index], variable == city_list[date_index+1]),  'eat in which city d')
                        s.assert_and_track(Implies(arrives[date_index]> 22, variable == city_list[date_index]),'eat in which city d')
                        s.assert_and_track(Implies(arrives[date_index]< 17, variable == city_list[date_index+1]),'eat in which city d')

                variables['restaurant_index'] = [Int('restaurant_{}_index'.format(i)) for i in range(3*7)]
                all_restaurant_price = 0
                
                for i, variable in enumerate(variables['restaurant_index']):
                    restaurant_city = variables['restaurant_in_which_city'][i]
                    restaurant_price_list, restaurant_list_length = RestaurantSearch.get_info(restaurant_price, restaurant_city, 'Price')
                    
                    # Assert valid index range
                    s.assert_and_track(Implies(restaurant_city != -1, And(variable >= 0, variable < restaurant_list_length)), 'valid restaurant index')
                    s.assert_and_track(Implies(restaurant_city == -1, variable == -1), 'valid restaurant index')
                    
                    # Assert non-repeating restaurants in the same city
                    for j in range(i-1, -1, -1):
                        s.assert_and_track(Implies(And(restaurant_city != -1, variables['restaurant_in_which_city'][j] == restaurant_city), 
                                                  variable != variables['restaurant_index'][j]), 'non repeating restaurant index')
                    
                    # Calculate price
                    all_restaurant_price += 2 * If(restaurant_city != -1, RestaurantSearch.get_info_for_index(restaurant_price_list, variable), 0)
                
                # Set cuisine type constraints for each required cuisine
                variables['cuisines_type'] = [Int('cuisines_' + i) for i in ['Mexican', 'Italian', 'Mediterranean', 'Indian']]
                for index, cuisine in enumerate(['Mexican', 'Italian', 'Mediterranean', 'Indian']):
                    count = 0
                    for i, variable in enumerate(variables['restaurant_index']):
                        restaurant_city = variables['restaurant_in_which_city'][i]
                        cuisines_list, _ = RestaurantSearch.get_info(restaurant_cuisines, restaurant_city, 'Cuisines')
                        count += If(RestaurantSearch.check_exists(cuisine, cuisines_list, variable), 1, 0)
                    s.assert_and_track(variables['cuisines_type'][index] == count, cuisine + 'type restaurant')
                    s.assert_and_track(variables['cuisines_type'][index] > 0, cuisine + 'type restaurant is visited')
attraction_info = AttractionSearch.run_for_all_cities(cities, city_list)
                variables['attraction_in_which_city'] = [Int('attraction_' + str(i)) for i in range(7)]
                for i, variable in enumerate(variables['attraction_in_which_city']):
                    s.assert_and_track(variable == If(arrives[i] > 18, city_list[i], If(i < len(city_list)-1, city_list[i+1], city_list[i])), 'attraction in which city')

                    s.assert_and_track(Implies(variables['attraction_in_which_city'][i] == -1, variable == -1), 'valid attraction index')
                    for j in range(i-1, -1, -1):
                        s.assert_and_track(Implies(And(variables['attraction_in_which_city'][i] != -1, variables['attraction_in_which_city'][i] == variables['attraction_in_which_city'][j]), variable != variables['attraction_index'][j]), 'non repeating attraction index')
# Run AccommodationSearch for each destination city  
                accommodation_info, accommodation_constraints = AccommodationSearch.run_for_all_cities(cities, [city_0, city_1, city_2])  
                # Set 'accommodation_index' variables for 3 (1 per city) accommodations  
                variables['accommodation_index'] = [Int('accommodation_{}_index'.format(i)) for i in range(3)]  
                # For each 'accommodation_index', get specific price info based on accommodation in which city variable, assert 'accommodation_index' variable are within valid range, calculate number of room need for 1 person and accommodation price  
                all_accommodation_price = 0  
                for i, variable in enumerate(variables['accommodation_index']):  
                    accommodation_price_list, accommodation_list_length = AccommodationSearch.get_info(accommodation_info, variables['city'][i], 'Price')

                    accommodation_maximum_occupancy_list, _ = AccommodationSearch.get_info(accommodation_info, variables['city'][i], 'Maximum_occupancy')

                    all_accommodation_price += (variables['departure_dates'][i+1] - variables['departure_dates'][i]) * num_room * AccommodationSearch.get_info_for_index(accommodation_price_list, variable)  
                # For each city, get accommodation minimum night info and assert it to be less than the days stay in this city  
                for index, city in enumerate(variables['city']):  
                    accommodation_minimum_nights_list, _ = AccommodationSearch.get_info(accommodation_info, city, 'Minimum_nights')

                    s.assert_and_track(minimum_night <= variables['departure_dates'][index+1]- variables['departure_dates'][index], 'minimum nights satisfied')  
                # For each 'accommodation_index', get specific room type and house rules info, assert 'Entire home/apt' exist for all accommodations, assert 'Shared room' does not exist for all accommodations, assert 'No parties' does not exist for all accommodations  
                for i, variable in enumerate(variables['accommodation_index']):
