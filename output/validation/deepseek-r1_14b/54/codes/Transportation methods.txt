variables['flight'] = [Bool('flight_' + str(i)) for i in range(3)]
                variables['self-driving'] = [Bool('self-driving_' + str(i)) for i in range(3)]
                variables['taxi'] = [Bool('taxi_' + str(i)) for i in range(3)]
                
                s.assert_and_track(Or(variables['flight'][0], variables['self-driving'][0], variables['taxi'][0]), 'either flight, self-driving, or taxi for first leg')
                s.assert_and_track(Or(variables['flight'][1], variables['self-driving'][1], variables['taxi'][1]), 'either flight, self-driving, or taxi for second leg')
                s.assert_and_track(Or(variables['flight'][2], variables['self-driving'][2], variables['taxi'][2]), 'either flight, self-driving, or taxi for third leg')
                
                s.assert_and_track(Not(And(variables['flight'][0], variables['self-driving'][0])), 'cannot use both flight and self-driving for first leg')
                s.assert_and_track(Not(And(variables['flight'][0], variables['taxi'][0])), 'cannot use both flight and taxi for first leg')
                s.assert_and_track(Not(And(variables['taxi'][0], variables['self-driving'][0])), 'cannot use both taxi and self-driving for first leg')
                
                s.assert_and_track(Not(And(variables['flight'][1], variables['self-driving'][1])), 'cannot use both flight and self-driving for second leg')
                s.assert_and_track(Not(And(variables['flight'][1], variables['taxi'][1])), 'cannot use both flight and taxi for second leg')
                s.assert_and_track(Not(And(variables['taxi'][1], variables['self-driving'][1])), 'cannot use both taxi and self-driving for second leg')
                
                s.assert_and_track(Not(And(variables['flight'][2], variables['self-driving'][2])), 'cannot use both flight and self-driving for third leg')
                s.assert_and_track(Not(And(variables['flight'][2], variables['taxi'][2])), 'cannot use both flight and taxi for third leg')
                s.assert_and_track(Not(And(variables['taxi'][2], variables['self-driving'][2])), 'cannot use both taxi and self-driving for third leg')
                
                s.assert_and_track(Implies(Or(variables['flight'][0], variables['flight'][1], variables['flight'][2]), Not(Or(variables['self-driving'][0], variables['self-driving'][1], variables['self-driving'][2]))), 'if any flight is used, self-driving cannot be used')
                s.assert_and_track(Implies(Or(variables['taxi'][0], variables['taxi'][1], variables['taxi'][2]), Not(Or(variables['self-driving'][0], variables['self-driving'][1], variables['self-driving'][2]))), 'if any taxi is used, self-driving cannot be used')
