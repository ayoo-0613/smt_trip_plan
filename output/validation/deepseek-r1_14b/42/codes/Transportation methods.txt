                variables['taxi'] = [Bool('taxi_' + str(i)) for i in range(4)]

                s.assert_and_track(Not(And(Not(variables['flight'][0]), Not(variables['self-driving'][0]), Not(variables['taxi'][0]))), 'transportation 1 must use a method')
