                driving_1_arrtime, _ = DistanceSearch.get_info(driving_info, variables['city'][0], variables['city'][1], 'Duration') # Use driving_1_arrtime, do not customize
                driving_2_arrtime, _ = DistanceSearch.get_info(driving_info, variables['city'][1], 'Bakersfield', 'Duration') # Use driving_2_arrtime, do not customize

attraction_info = AttractionSearch.run_for_all_cities(cities, cities)

variables['budget_limit'] = RealVal(6100)
