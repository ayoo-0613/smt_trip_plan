variables['flight'] = [Bool('flight_travel_' + str(i)) for i in range(6)]
                variables['self-driving'] = [Bool('self-driving_travel_' + str(i)) for i in range(6)]
                variables['taxi'] = [Bool('taxi_travel_' + str(i)) for i in range(6)]
                
                # Assert only one of flight, self-driving, or taxi is used for each transportation leg
                for i in range(6):
                    s.assert_and_track(Or(variables['flight'][i], variables['self-driving'][i], variables['taxi'][i]),  f'either flight, self-driving, or taxi for transportation {i+1}')
                    s.assert_and_track(Not(And(variables['flight'][i], variables['self-driving'][i])), f'flight and self-driving not simultaneously for transportation {i+1}')
                    s.assert_and_track(Not(And(variables['flight'][i], variables['taxi'][i])), f'flight and taxi not simultaneously for transportation {i+1}')
                    s.assert_and_track(Not(And(variables['self-driving'][i], variables['taxi'][i])), f'self-driving and taxi not simultaneously for transportation {i+1}')
                
                # Ensure all 6 transportations are feasible based on the selected method
                s.assert_and_track(Implies(Or(*variables['flight']), Not(Or(*variables['self-driving']))), 'no self-driving if taken flight for any transportation')
                s.assert_and_track(Implies(Or(*variables['taxi']), Not(Or(*variables['self-driving']))), 'no self-driving if taken taxi for any transportation')
