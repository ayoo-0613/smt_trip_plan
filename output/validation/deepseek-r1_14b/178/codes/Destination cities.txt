cities = ['Austin', 'Houston', 'San Antonio']

# Loop through all possible combinations of city indices for three cities
for city_0_index, _ in enumerate(cities):
    for city_1_index, _ in enumerate(cities):
        for city_2_index, _ in enumerate(cities):

            s = Optimize()
