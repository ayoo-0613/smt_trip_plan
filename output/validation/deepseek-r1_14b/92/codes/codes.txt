                s.assert_and_track(Not(And(variables['self-driving'][i], variables['taxi'][i])), f'self-driving and taxi not simultaneously for transportation {i+1}')
            
            # Assert no self-driving if any flight or taxi is used
            s.assert_and_track(Implies(Or(*variables['flight']), Not(Or(*variables['self-driving']))), 'no self-driving if taken flight for any transportation')
            s.assert_and_track(Implies(Or(*variables['taxi']), Not(Or(*variables['self-driving']))), 'no self-driving if taken taxi for any transportation')

flight_info = FlightSearch.run_for_all_cities_and_dates('New Orleans', cities, [city_0, city_1], ["2022-03-12", "2022-03-13", "2022-03-14", "2022-03-15", "2022-03-16"]) # use flight_info, do not customize

            flight_3_price_list, flight_3_price_list_length = FlightSearch.get_info(flight_info, variables['city'][2], 'New Orleans', variables['departure_dates'][3], 'Price')

            s.assert_and_track(Implies(variables['flight'][2], And(variables['flight_index'][2] >= 0,variables['flight_index'][2] < flight_2_price_list_length)), 'valid flight index for flight 2')

            s.assert_and_track(Implies(Not(variables['flight'][2]), variables['flight_index'][2] == -1), 'valid flight index for flight 2')

            driving_3_distance, driving_3_length = DistanceSearch.get_info(driving_info, variables['city'][2], 'New Orleans', 'Distance')

            s.assert_and_track(Implies(Or(variables['self-driving'][2], variables['taxi'][2]), driving_2_length > 0), 'driving is possible for transportation 2')  
            s.assert_and_track(Implies(Or(variables['self-driving'][3], variables['taxi'][3]), driving_3_length > 0), 'driving is possible for transportation 3')

            self_driving_1_price = 0.05 * driving_1_distance * math.ceil(3 / 5)  
            self_driving_2_price = 0.05 * driving_2_distance * math.ceil(3 / 5)  
            self_driving_3_price = 0.05 * driving_3_distance * math.ceil(3 / 5)

            taxi_1_price = driving_1_distance * math.ceil(3 / 4)  
            taxi_2_price = driving_2_distance * math.ceil(3 / 4)  
            taxi_3_price = driving_3_distance * math.ceil(3 / 4)

            driving_2_arrtime, _ = DistanceSearch.get_info(driving_info, variables['city'][1], variables['city'][2], 'Duration')  # Use driving_2_arrtime, do not customize  
            driving_3_arrtime, _ = DistanceSearch.get_info(driving_info, variables['city'][2], 'New Orleans', 'Duration')  # Use driving_3_arrtime, do not customize

            transportation_0_arrtime = If(variables['flight'][0], flight_0_arrtime, driving_0_arrtime)

                    s.assert_and_track(Implies(arrives[date_index] < 10, variable == city_list[date_index+1]), 'eat in which city l')

# Run AttractionSearch to get attraction info for two Florida cities
