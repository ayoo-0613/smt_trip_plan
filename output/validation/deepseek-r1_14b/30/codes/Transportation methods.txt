variables['flight'] = [Bool('flight_travel_' + str(i)) for i in range(2)]
            variables['self-driving'] = [Bool('self-driving_travel_' + str(i)) for i in range(2)]
            variables['taxi'] = [Bool('taxi_travel_' + str(i)) for i in range(2)]
            
            s.assert_and_track(ExactlyOne([variables['flight'][0], variables['self-driving'][0], variables['taxi'][0]]), 'exactly one transportation method for first travel')
            s.assert_and_track(ExactlyOne([variables['flight'][1], variables['self-driving'][1], variables['taxi'][1]]), 'exactly one transportation method for second travel')
            
            s.assert_and_track(Implies(variables['flight'][0], Not(variables['self-driving'][0])), 'no self-driving if flight used in first transportation')
            s.assert_and_track(Implies(variables['flight'][1], Not(variables['self-driving'][1])), 'no self-driving if flight used in second transportation')
            
            s.assert_and_track(Implies(variables['taxi'][0], Not(variables['self-driving'][0])), 'no self-driving if taxi used in first transportation')
            s.assert_and_track(Implies(variables['taxi'][1], Not(variables['self-driving'][1])), 'no self-driving if taxi used in second transportation')
            
            s.assert_and_track(Not(variables['self-driving'][0]), 'no self-driving for first transportation')
            s.assert_and_track(Not(variables['self-driving'][1]), 'no self-driving for second transportation')
